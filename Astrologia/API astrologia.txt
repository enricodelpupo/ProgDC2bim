using Microsoft.AspNetCore.Mvc;
using System.Collections.Generic;

namespace AstroService.Controllers
{
    [Route("api/[controller]")]
    [ApiController]
    public class ControllerAstrologia: ControllerBase
    {
        private readonly ServicoAstrologia_astrologyService;

        public AstrologyController(ServicoAstrologia astrologyService)
        {
            _astrologyService = astrologyService;
        }

        [HttpGet("{sign}")]
        public MensagemDoDia MensagemDoDia(string sign)
        {
            return _astrologyService.GetMensagemDoDia(sign);
        }

        [HttpGet("{sign}/advanced")]
        public InfoAstrologia GetInfoAstrologia(string sign)
        {
            return _astrologyService.GetInfoAstrologia(sign);
        }
    }
}
